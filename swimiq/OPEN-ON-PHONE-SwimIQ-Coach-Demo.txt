===== SCRIPT =====

# SwimIQ Coach Demo — finds StrokeIQ anywhere on Desktop/OneDrive/Documents
# Save as Desktop\SwimIQ-Coach-Demo.ps1  (All files)
# Right-click → Run with PowerShell

$ErrorActionPreference = 'Continue'
Write-Host 'Looking for swimiq\pubspec.yaml ...' -ForegroundColor Cyan

$searchRoots = @(
  "$env:USERPROFILE\Desktop",
  "$env:USERPROFILE\OneDrive\Desktop",
  "$env:OneDrive\Desktop",
  "$env:USERPROFILE\Documents",
  "$env:USERPROFILE\OneDrive\Documents",
  'C:\SwimIQWork'
) | Where-Object { $_ -and (Test-Path $_) } | Select-Object -Unique

$swimiq = $null
foreach ($r in $searchRoots) {
  $hit = Get-ChildItem -Path $r -Filter 'pubspec.yaml' -Recurse -ErrorAction SilentlyContinue |
    Where-Object {
      $_.Directory.Name -eq 'swimiq' -and
      (Test-Path (Join-Path $_.DirectoryName 'lib\main.dart'))
    } |
    Select-Object -First 1
  if ($hit) {
    $swimiq = $hit.DirectoryName
    break
  }
}

if (-not $swimiq) {
  Write-Host 'ERROR: Could not find swimiq\pubspec.yaml automatically.' -ForegroundColor Red
  Write-Host ''
  Write-Host 'Do this:'
  Write-Host '  1) File Explorer → search This PC for: pubspec.yaml'
  Write-Host '  2) Open that file location'
  Write-Host '  3) Copy the address bar path (should end in \swimiq)'
  Write-Host '  4) Paste it below'
  Write-Host ''
  $typed = Read-Host 'Paste full path to swimiq folder'
  $typed = $typed.Trim().Trim('"')
  if ($typed -and (Test-Path (Join-Path $typed 'pubspec.yaml'))) {
    $swimiq = $typed
  } else {
    Write-Host 'That path is not a swimiq folder.' -ForegroundColor Red
    Read-Host 'Press Enter'
    exit 1
  }
}

$root = Split-Path $swimiq -Parent
$elite = Join-Path $root 'services\video_analysis'
Write-Host "FOUND swimiq: $swimiq" -ForegroundColor Green
Write-Host "Repo root:    $root"

if (-not (Test-Path $elite)) {
  Write-Host "ERROR: Missing Elite folder: $elite" -ForegroundColor Red
  Write-Host 'Your StrokeIQ copy is incomplete (needs services\video_analysis).'
  Read-Host 'Press Enter'
  exit 1
}

$eliteCmd = @"
`$ErrorActionPreference='Continue'
Set-Location '$elite'
Write-Host 'Elite — KEEP THIS WINDOW OPEN' -ForegroundColor Cyan
if (Test-Path '.\.venv\Scripts\python.exe') { `$py = '.\.venv\Scripts\python.exe' } else { `$py = 'python' }
& `$py -m uvicorn app.main:app --host 127.0.0.1 --port 8080
"@
Start-Process powershell -ArgumentList @('-NoExit','-ExecutionPolicy','Bypass','-Command', $eliteCmd)

Write-Host 'Waiting for Elite...'
$ready = $false
for ($i = 1; $i -le 40; $i++) {
  Start-Sleep -Seconds 3
  try {
    $r = Invoke-WebRequest -UseBasicParsing -Uri 'http://127.0.0.1:8080/health' -TimeoutSec 2
    if ($r.StatusCode -eq 200) { $ready = $true; break }
  } catch {}
  Write-Host "  try $i/40..."
}
if (-not $ready) {
  Write-Host 'Elite failed — read the other window. Try: winget install Gyan.FFmpeg' -ForegroundColor Yellow
  Read-Host 'Press Enter'
  exit 1
}

Set-Location $swimiq
if (-not (Test-Path '.\.env')) {
  Write-Host 'ERROR: swimiq\.env missing (Supabase keys).' -ForegroundColor Red
  Read-Host 'Press Enter'
  exit 1
}

$envMap = @{}
Get-Content '.\.env' | ForEach-Object {
  if ($_ -match '^\s*#' -or $_ -notmatch '=') { return }
  $k,$v = $_.Split('=',2)
  $envMap[$k.Trim()] = $v.Trim().Trim('"').Trim("'")
}
$url = $envMap['SUPABASE_URL']
$key = $envMap['SUPABASE_ANON_KEY']
$api = $envMap['ANALYSIS_API_BASE_URL']
if ([string]::IsNullOrWhiteSpace($api)) { $api = 'http://127.0.0.1:8080' }
$v2 = $envMap['VIDEO_ENGINE_V2']
if ([string]::IsNullOrWhiteSpace($v2)) { $v2 = 'true' }

Write-Host 'Starting Chrome on 127.0.0.1 ...'
flutter pub get
flutter run -d chrome `
  --dart-define="SUPABASE_URL=$url" `
  --dart-define="SUPABASE_ANON_KEY=$key" `
  --dart-define="VIDEO_ENGINE_V2=$v2" `
  --dart-define="ANALYSIS_API_BASE_URL=$api" `
  --dart-define=VIDEO_ENGINE_V2_DUAL_RUN=false
